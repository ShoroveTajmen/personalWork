for(var i = 0; i<3; i++){
document.querySelectorAll(".button-S")[i].addEventListener("click",function(){
  var text = this.innerHTML;
  console.log(text);
  audioPlay(text);
})
}

function audioPlay(text){
  switch(text){
    case "a":
      var audio = new Audio("./mp3/2.mp4");
      audio.play();
      break;
  }
}
