// letter grade

var marks = parseFloat(prompt("Enter a number: "))

if(marks >= 80)
    console.log("A+")
else if(marks >= 70)
    log("A")
else if(marks >= 60)
    console.log("A-")
else if(marks >= 50)
    console.log("B")
else if(marks >= 40)
    console.log("c")
else if(marks >= 30)
    console.log("C-")
else
    console.log("Fail")
