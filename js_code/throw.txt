## JS CODE ##

document.querySelector("#check1").addEventListener("click",function(){
  var num = document.querySelector("#text1").value;
  

  try{
    if(num < 5){
      throw "input is too low"
    }
    else if(num > 10){
      throw "input is too high"
    }
  }catch(err){
      console.log(err);
      // document.write(err);

  }
})
