 //default parameter

" use strict "
function  message(text = "hello this is default") 
{
  console.log(`${text}`)
  
}
message();
message("i love js es6");





//rest parameter

" use strict "
function  message(x,y ,...z) 
{
  console.log(`x= ${x}, y=${y},z = ${z}`);
  
}
message(10,20,30,40,50);
