//celsius to fahrenheit

var cels = parseFloat(prompt("Enter celsius : ")); 
var farn = (cels*9/5) + 32;
document.write("celsius is : " + farn);




//fahrenheit to celsius

var fahr = parseFloat(prompt("Enter fahrenheit : ")); 
var cels = (fahr - 32) * (5/9) ;
document.write("celsius is : " + cels);
