var img = ["./images/img1.jpg","./images/img2.jpg",
"./images/img3.jpg","./images/img4.jpg","./images/img5.jpg"];

var imgtag = document.querySelector("img");
var count = 0;


function next(){
    count++;
    if(count >= img.length){
        count = 0;
        imgtag.src = img[count];

    }
    else{
         imgtag.src = img[count];
    }
   

}
function prev(){
    count--;
    if(count<0){
        count = img.length-1;
        imgtag.src = img[count];
    }
    else{
        
    imgtag.src = img[count];
    }

}
